# 1 완주하지 못한 선수 (javascript)

# 문제 설명

수많은 마라톤 선수들이 마라톤에 참여하였습니다. 단 한 명의 선수를 제외하고는 모든 선수가 마라톤을 완주하였습니다.

마라톤에 참여한 선수들의 이름이 담긴 배열 participant와 완주한 선수들의 이름이 담긴 배열 completion이 주어질 때, 완주하지 못한 선수의 이름을 return 하도록 solution 함수를 작성해주세요.

# 제한사항

마라톤 경기에 참여한 선수의 수는 1명 이상 100,000명 이하입니다. completion의 길이는 participant의 길이보다 1 작습니다. 참가자의 이름은 1개 이상 20개 이하의 알파벳 소문자로 이루어져 있습니다. 참가자 중에는 동명이인이 있을 수 있습니다.

# 입출력 예

participant completion return [leo, kiki, eden] [eden, kiki] leo [marina, josipa, nikola, vinko, filipa] [josipa, filipa, marina, nikola] vinko [mislav, stanko, mislav, ana] [stanko, ana, mislav] mislav

# 입출력 예 설명

예제 #1 leo는 참여자 명단에는 있지만, 완주자 명단에는 없기 때문에 완주하지 못했습니다.

예제 #2 vinko는 참여자 명단에는 있지만, 완주자 명단에는 없기 때문에 완주하지 못했습니다.

예제 #3 mislav는 참여자 명단에는 두 명이 있지만, 완주자 명단에는 한 명밖에 없기 때문에 한명은 완주하지 못했습니다.

# 정확성

const solution = (participant, completion) => {
  completion.forEach((string) => {
    const index = participant.indexOf(string)
    participant.splice(index, 1)
  });

  return participant[0];
}

# 효율성

const solution = (participant, completion) => {
  let answer = '';
  const merged = participant.concat(completion);
  const sorted = merged.sort();

  for (let i = 0; i < sorted.length; i += 2) {
    const before = sorted[i];
    const after = sorted[i + 1];

    if (before !== after || i === sorted.length) {
      answer = before;
      break;
    }
  }

  return answer;
}

def solution(phone_book):
  answer = True
  sorted_phone_book = sorted(phone_book, key=len)
  for index, value in enumerate(sorted_phone_book):
    for i in range(index, len(phone_book)):
      if i == index:
        continue
      if value == sorted_phone_book[i][:len(value)]:
        answer = False
        break
    break
  return answer

# 3 위장 (javascript)

const solution = (clothes) => {
  const clothKinds = clothes.map((cloth) => cloth[1]).sort();
  const countByKind = clothKinds.reduce((result, kind) => {
    if (result.hasOwnProperty(kind)) {
      result[kind] += 1;
    } else {
      // 입지않은 경우의 수 포함(생각도 못함)
      result[kind] = 2;
    }
    return result;
  }, {});
  const counts = Object.values(countByKind);

  // 아무것도 입지않은 경우의 수 제외(생각도 못함)
  const answer = counts.reduce((total, v) => total *= v) - 1
  return answer;
}
